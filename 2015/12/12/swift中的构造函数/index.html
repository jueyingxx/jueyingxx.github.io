swift中的构造函数

发表于 2015-12-12   |  

`import UIKit

class Person: NSObject {
// 存储属性，开辟空间，保存数据
// OC中，alloc分配空间， init进行初始化
// swift中，要求必须给非Optional的属性赋初始值
var name: String
var age: Int

// 构造函数，没有func
// override重写父类
// 集成自父类NSObjec，默认提供init()函数
override init() {
    name = "jueying"
    age = 18
}

// 重载函数，没有返回值，可以增加参数
// 外部调用，传入参数进行初始化
// 如果自己创建构造函数，但是没有重写init(), 系统会取消init()函数
// 外部调用没有办法直接用Person()创建对象
init(name: String, age: Int) {
    self.name = name
    self.age = age
}

}`
